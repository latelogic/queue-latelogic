Integration in Minutes

Node.js SDK 

import { FlightQueue } from 'flightqueue-sdk';

const client = new FlightQueue('sk_live_...');

// Subscribe to airport departures
client.stream.subscribe('departures', {
  airport: 'JFK'
});

client.stream.on('flight_departure', (flight) => {
  console.log(`Flight ${flight.id} departed!`);
});

Python SDK 

import flightqueue

client = flightqueue.Client('sk_live_...')

# Get historical flight data
history = client.flights.history(
    flight_id='AA100',
    date='2025-12-25'
)

print(f"Flight duration: {history.duration_minutes}")
